pwd - просмотр текущей директории 
cd - переход в выбранную директорию
~ - символ домашней директории
ls - содержание директории
ls -a - содержание со всеми скрытыми файлами
ls~ содержание домашней директории
ls.. содержании директории на шаг выше
touch - создание файла
mkdir - создание директории
mkdir -p - создание стопки директорий
cp - копирование файлов/папок
mv - перемещение папок/файлов
cat - чтение текстовых файлов
rm - удаление файлов
rm -r - удаление папки с файлами
rmdr - удаление дирктории
&& - and как в java (несколько функции сразу)
git init - создание репрезитория
rm -rf .git - удаление репрезитория
git status - статус репрезитория
git add - добавить в репрезиторий
git commit -m - коммит
git commit - переход в программу vi, выход : :q!
git log - история коммитов
git clone - копирование репрезитория
